# Build the client
FROM node:20.11.1 AS client-build
WORKDIR /app/client
COPY client/package*.json ./
RUN npm install
COPY client/ ./
RUN npm run build

# Build the server
FROM node:20.11.1 AS server-build
WORKDIR /app/server
COPY server/package*.json ./
RUN npm install
COPY server/ ./

# Final stage
FROM node:20.11.1
WORKDIR /app
COPY --from=client-build /app/client/build ./client/build
COPY --from=server-build /app/server ./

# Set the environment variables
ENV NODE_ENV=production

# Expose the port for the server
EXPOSE 3000

# Start the server
CMD ["npm", "start"]
